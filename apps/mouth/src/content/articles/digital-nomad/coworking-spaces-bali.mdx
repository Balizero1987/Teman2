  title="Coworking Budget"
  description="Estimate your monthly workspace costs"
  fields={[
    {

      label: "How Often Will You Use It?",
      type: "select",

      ],

    },
    {
      id: "tier",
      label: "Space Quality",
      type: "select",
      options: [
        { value: "budget", label: "Budget friendly" },
        { value: "standard", label: "Standard coworking" },
        { value: "premium", label: "Premium/private office" }
      ],
      defaultValue: "standard"
    }
  ]}
  calculateResult={(values) => {
    // Pricing estimates
    const pricing = {
      budget: { monthly: 1500000, day: 100000 },
      standard: { monthly: 2500000, day: 150000 },
      premium: { monthly: 4000000, day: 250000 }
    };

    const tier = pricing[values.tier];
    let monthly;

    switch (values.frequency) {
      case "daily":
        monthly = tier.monthly;
        break;
      case "parttime":
        monthly = Math.min(tier.day * 16, tier.monthly);
        break;
      case "occasional":
        monthly = tier.day * 8;
        break;
      case "dropins":
        monthly = tier.day * 4;
        break;
    }

    return {
      total: monthly,
      breakdown: [
        { label: "Workspace Cost", amount: monthly },
        { label: "Coffee/Drinks", amount: 300000 },
        { label: "Occasional Lunch", amount: 500000 }
      ],
      currency: "IDR",
      note: `Approximately USD ${Math.round(monthly / 16000)}/month`
    };
  }}

## Tips for Digital Nomads

1. **Try before committing** - Buy day passes first
2. **Check backup internet** - Ask what happens when main fails
3. **Visit during your work hours** - Check if it's too busy
4. **Test video call quality** - Essential for most remote work
5. **Consider location** - Near your accommodation saves time
6. **Use the community** - Networking is valuable
7. **Have mobile backup** - Telkomsel SIM for emergencies
8. **Book meeting rooms early** - Popular spaces fill up
